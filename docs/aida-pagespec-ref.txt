spec := <token> ( "," <token> )*

token := <num> | <range>

num := [ "^" ] ( "0".."9" )+

range := [ "*" | <num> "-" <num> ] [ "{" <condition> "}" ] [ <exclude> ]
      := <condition> [ <exclude> ]     // = "*{" <condition> "}"
      := <exclude>                     // = "*" <exclude>

exclude := "!" <token>

condition := <keyword> [ <op> <condition> ]

keyword := "odd" | "even"

op := "or" | "|" | "and" | "&"
